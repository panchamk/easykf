00159         inline void ukf_scalar_iterate(ukf_param &p, ukf_scalar_state &s, FunctObj g, gsl_vector * xk, double dk)
00160         {
00161             // Here, we implement the UKF for parameter estimation in the scalar case
00162             // The notations follow p93 of the PhD thesis of Van Der Merwe, "Sigma-Point Kalman Filters for Probabilistic Inference in Dynamic State-Space Models"
00163 
00164             // ************************************************** //
00165             // ************ Time update equations    ************ //
00166             // ************************************************** //
00167             // Add the evolution noise to the parameter covariance Eq 3.137
00168             gsl_matrix_add(s.Pk, s.Prrk);
00169 
00170             // ************************************************** //
00171             // ************ Compute the sigma points ************ //
00172             // ************************************************** //
00173             // Equations 3.138
00174             // w_k^j = w_(k-1)  <-- this is here denoted s.w
00175             // w_k^j = w_(k-1) + gamma Sk_j for 1 <= j <= n
00176             // w_k^j = w_(k-1) - gamma Sk_j for n+1 <= j <= 2n
00177 
00178             // Perform a cholesky decomposition of Pk
00179             gsl_matrix_memcpy(s.Sk, s.Pk);
00180             gsl_linalg_cholesky_decomp(s.Sk);
00181             // Set all the elements of Lpi strictly above the diagonal to zero
00182             for(int k = 0 ; k < p.n ; k++)
00183                 for(int j = 0 ; j < k ; j++)
00184                     gsl_matrix_set(s.Sk,j,k,0.0);
00185 
00186             gsl_matrix_set_col(s.sigmaPoints,0, s.w);
00187             for(int j = 1 ; j < p.n+1 ; ++j)
00188                 for(int i = 0 ; i < p.n ; ++i)
00189                     gsl_matrix_set(s.sigmaPoints, i, j, gsl_vector_get(s.w, i) + p.gamma * gsl_matrix_get(s.Sk,i,j-1));
00190 
00191             for(int j = p.n+1 ; j < p.nbSamples ; ++j)
00192                 for(int i = 0 ; i < p.n ; ++i)
00193                     gsl_matrix_set(s.sigmaPoints, i, j, gsl_vector_get(s.w, i) - p.gamma * gsl_matrix_get(s.Sk,i,j-(p.n+1)));
00194 
00195             // ************************************************** //
00196             // ***** Compute the images of the sigma points ***** //
00197             // ************************************************** //
00198 
00199             // Compute the images of the sigma points
00200             // and the mean of the yj
00201             s.d_mean = 0.0;
00202             gsl_vector_set_zero(s.dk);
00203             for(int j = 0 ; j < p.nbSamples ; j++)
00204             {
00205                 // Equation 3.139
00206                 gsl_matrix_get_col(s.wk, s.sigmaPoints,j);
00207                 s.dk->data[j] = g(s.wk,xk);
00208 
00209                 // Update the mean : y_mean = sum_[j=0..2n] wm_j y_j
00210                 // Equation 3.140
00211                 s.d_mean += s.wm->data[j] * s.dk->data[j];
00212             }
00213 
00214             // ************************************************** //
00215             // ************** Update the statistics ************* //
00216             // ************************************************** //
00217 
00218             gsl_vector_set_zero(s.Pwdk);
00219             // The covariance of the output is initialized with the observation noise covariance
00220             // Eq 3.142
00221             s.Pddk = s.Peek;
00222             for(int j = 0 ; j < p.nbSamples ; j++)
00223             {
00224                 // Eq 3.142
00225                 s.Pddk += s.wc->data[j] * gsl_pow_2(s.dk->data[j] - s.d_mean);
00226                 // Eq 3.143
00227                 for(int i = 0 ; i < p.n ; ++i)
00228                 {
00229                     s.Pwdk->data[i] = s.Pwdk->data[i] + s.wc->data[j] * (gsl_matrix_get(s.sigmaPoints,i,j) - s.w->data[i]) * (s.dk->data[j] - s.d_mean) ;
00230                 }
00231             }
00232 
00233             // ************************************************** //
00234             // ******* Kalman gain and parameters update ******** //
00235             // ************************************************** //
00236 
00237             //if(s.Pddk == 0.0)
00238             //    printf("[Error] Output covariance is null !");
00239             // May not occur as soon as the observation noise covariance is set != 0.0
00240 
00241             // Eq. 3.144
00242             for(int i = 0 ; i < p.n ; ++i)
00243                 s.Kk->data[i] = s.Pwdk->data[i] / s.Pddk;
00244 
00245             // Eq 3.145
00246             // wk = w_(k-1) + Kk * (dk - d_mean)
00247             s.ino_dk = dk - s.d_mean;
00248             for(int i = 0 ; i < p.n ; ++i)
00249                 s.w->data[i] = s.w->data[i] + s.Kk->data[i] * s.ino_dk;
00250 
00251             // Eq. 3.146
00252             // Pk = P_(k-1) - Pddk . Kk Kk^T
00253             for(int i = 0 ; i < p.n ; ++i)
00254             {
00255                 gsl_matrix_set(s.Kk_mat, i, 0, s.Kk->data[i]);
00256                 gsl_matrix_set(s.Kk_mat_T, 0, i, s.Kk->data[i]);
00257             }
00258             gsl_blas_dgemm(CblasNoTrans,CblasNoTrans,-s.Pddk, s.Kk_mat, s.Kk_mat_T,1.0,s.Pk);
00259 
00260             // Update of the evolution noise
00261             p.evolution_noise->updateEvolutionNoise(p,s);
00262         }

00270         {
00271             // ************************************************** //
00272             // ************ Compute the sigma points ************ //
00273             // ************************************************** //
00274 
00275             // 1- Compute the cholesky decomposition of Pk
00276             gsl_matrix_memcpy(s.temp_n_n, s.Pk);
00277             gsl_linalg_cholesky_decomp(s.temp_n_n);
00278 
00279             // 2 - Set all the elements of Sk_temp strictly above the diagonal to zero
00280             for(int k = 0 ; k < p.n ; k++)
00281                 for(int j = 0 ; j < k ; j++)
00282                     gsl_matrix_set(s.temp_n_n,j,k,0.0);
00283             // Now Sk_temp is a lower triangular matrix containing the cholesky decomposition of Pk
00284 
00285             // 3- Compute the sigma points
00286             // Equations 3.138
00287             // w_k^j = w_(k-1)  <-- this is here denoted s.w
00288             // w_k^j = w_(k-1) + gamma Sk_j for 1 <= j <= n
00289             // w_k^j = w_(k-1) - gamma Sk_j for n+1 <= j <= 2n
00290             gsl_matrix_set_col(s.sigmaPoints,0, s.w);
00291             for(int j = 1 ; j < p.n+1 ; j++)
00292                 for(int i = 0 ; i < p.n ; ++i)
00293                     gsl_matrix_set(s.sigmaPoints, i, j, gsl_vector_get(s.w, i) + p.gamma * gsl_matrix_get(s.temp_n_n,i,j-1));
00294 
00295             for(int j = p.n+1 ; j < p.nbSamples ; j++)
00296                 for(int i = 0 ; i < p.n ; ++i)
00297                     gsl_matrix_set(s.sigmaPoints, i, j, gsl_vector_get(s.w, i) - p.gamma * gsl_matrix_get(s.temp_n_n,i,j-(p.n+1)));
00298 
00299             // ************************************************** //
00300             // ***** Compute the images of the sigma points ***** //
00301             // ************************************************** //
00302 
00303             // Compute the images of the sigma points
00304             // and their mean
00305             dk = 0.0;
00306             for(int j = 0 ; j < p.nbSamples ; j++)
00307             {
00308                 gsl_matrix_get_col(s.wk, s.sigmaPoints,j);
00309                 // Update the mean : d_mean = sum_[j=0..2n] w_j y_j
00310                 dk += gsl_vector_get(s.wm,j) * g(s.wk,xk);
00311             }
00312         }
