00068         void choleskyUpdate(gsl_matrix * sigmaTheta, double alpha, gsl_vector *x) {
00069 
00070             /*
00071                 This function performs a cholesky update of the
00072                 cholesky factorization sigmaTheta, that is it replaces
00073                 the Cholesky factorization sigmaTheta by the cholesky factorization of
00074 
00075                             sigmaTheta*sigmaTheta^T + alpha * x * x^T
00076 
00077                 The algorithm is an adaptation of a LU factorization rank one update.
00078 
00079                 Reference is :
00080 
00081                 Peter Strange, Andreas Griewank and Matthias Bollhöfer.
00082                 On the Efficient Update of Rectangular LU Factorizations subject to Low Rank Modifications.
00083                 Electronic Transactions on Numerical Analysis, 26:161-177, 2007.
00084                 alg. is given in left part of fig.2.1.
00085 
00086                 Perhaps a more efficient algorithm exists, however it should do the work for now. And the code is probably not optimal...
00087             */
00088 
00089             if (sigmaTheta->size1 != sigmaTheta->size2)
00090                 std::cout<<"ERROR ukf::math::choleskyUpdate Cannot use CholeskyUpdate on non-squared matrices"<<std::endl ;
00091 
00092             int i,j,n= sigmaTheta->size1 ;
00093             double tmp ;
00094             gsl_matrix *U = gsl_matrix_alloc(n,n) ;
00095             gsl_vector *D = gsl_vector_alloc(n) ;
00096             gsl_vector *y = gsl_vector_alloc(n) ;
00097 
00098             // A first thing is to set SS' (chol factor) in a LU form, L being unitriangular
00099             // Compute U = L^T and D = diag(L)
00100             gsl_matrix_set_zero(U) ;
00101             for (i=0; i<n; i++){
00102                 gsl_vector_set(D,i,gsl_matrix_get(sigmaTheta,i,i));
00103                 for (j=0; j<=i; j++){
00104                     gsl_matrix_set(U,j,i,gsl_matrix_get(sigmaTheta,i,j));
00105                 }
00106             }
00107             // Replace L by L*D^{-1} and U by D*U
00108             for (i=0; i<n; i++){
00109                 for (j=0; j<=i; j++){
00110                     tmp = gsl_matrix_get(sigmaTheta,i,j);
00111                     tmp /= gsl_vector_get(D,j);
00112                     gsl_matrix_set(sigmaTheta,i,j,tmp);
00113                     tmp = gsl_matrix_get(U,j,i);
00114                     tmp *= gsl_vector_get(D,j);
00115                     gsl_matrix_set(U,j,i,tmp);
00116                 }
00117             }
00118 
00119             // compute the y = alpha x vector
00120             gsl_vector_memcpy(y,x) ;
00121             gsl_vector_scale(y,alpha) ;
00122 
00123             // perform the rank 1 LU modification
00124             for (i=0; i<n; i++){
00125 
00126                 // diagonal update
00127                 tmp = gsl_matrix_get(U,i,i) +
00128                       gsl_vector_get(x,i)*gsl_vector_get(y,i);
00129                 gsl_matrix_set(U,i,i,tmp);
00130                 tmp = gsl_vector_get(y,i);
00131                 tmp /= gsl_matrix_get(U,i,i);
00132                 gsl_vector_set(y,i,tmp);
00133 
00134                 for (j=i+1; j<n; j++){
00135                     // L (that is sigmaTheta) update
00136                     tmp = gsl_vector_get(x,j) -
00137                           gsl_vector_get(x,i)*gsl_matrix_get(sigmaTheta,j,i);
00138                     gsl_vector_set(x,j,tmp) ;
00139                     tmp = gsl_matrix_get(sigmaTheta,j,i) + gsl_vector_get(y,i) * gsl_vector_get(x,j);
00140                     gsl_matrix_set(sigmaTheta,j,i,tmp);
00141                 }
00142 
00143                 for (j=i+1; j<n; j++) {
00144                     // U update
00145                     tmp = gsl_matrix_get(U,i,j) +
00146                           gsl_vector_get(x,i)*gsl_vector_get(y,j);
00147                     gsl_matrix_set(U,i,j,tmp) ;
00148                     tmp = gsl_vector_get(y,j) -
00149                           gsl_vector_get(y,i) * gsl_matrix_get(U,i,j);
00150                     gsl_vector_set(y,j,tmp) ;
00151                 }
00152             }
00153 
00154             // Now we want the chol decomposition
00155             // first D = sqrt(diag(U)) ;
00156             for (i=0; i<n; i++){
00157                 tmp =  gsl_matrix_get(U,i,i) ;
00158 
00159                 if (tmp<=0){std::cout<< "WARNING ukf::math::choleskyUpdate::matrix not positive definite  : " << tmp << std::endl;}
00160                 gsl_vector_set(D,i,sqrt(tmp)) ;
00161             }
00162             // then L = L*D ;
00163             for (i=0; i<n; i++) {
00164                 for (j=0; j<n; j++) {
00165                     tmp = gsl_matrix_get(sigmaTheta,i,j) * gsl_vector_get(D,j);
00166                     gsl_matrix_set(sigmaTheta,i,j,tmp);
00167                 }
00168             }
00169             // that's all folks !
00170 
00171             //free memory
00172             gsl_matrix_free(U);
00173             gsl_vector_free(y);
00174             gsl_vector_free(D);
00175         }
